LABEL GeneraMapa
nmovs=0 : npush=0
for y=1 to MAXROW
for x=1 to MAXCOL
Mapa$(y,x)=Ground$
Dirty(y,x)=1
next x:next y

playerposx=-1:playerposy=-1
for index=1 to len(niveldata$(nivel))
temp$=mid$( niveldata$(nivel), index, 1)
if temp$=newline$ then 
   rows=rows+1
   if rows > MAXROW then
      retError$="ErrTOMUCHROWS"
   else
      if j> cols then cols=j fi
      j=1
   fi
else 
   opt=instr(valid$,temp$)
   if opt>0 then
      if temp$=player$ or temp$=playerstore$ then
         playerposy=rows
         playerposx=j
      fi
      if temp$=save$ then
         savepack=savepack+1
         packets=packets+1
      fi
      if temp$=packet$ then packets=packets+1 
      fi
      Mapa$(rows,j)=temp$
      j=j+1
      if j>MAXCOL then retError$="ErrTOMUCHCOLS" 
      fi
   else 
      retError$="ErrILLCHAR"  
   fi
fi
next index
